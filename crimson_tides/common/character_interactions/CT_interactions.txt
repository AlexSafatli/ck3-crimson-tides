sire_interaction = {
	interface_priority = 61
	use_diplomatic_range = yes
	category = interaction_category_friendly

  desc = sire_interaction_desc
	
	auto_accept = yes
	is_shown = {
    scope:actor = {
      has_trait = vampirism
    }
    scope:recipient = { 
      NOT = { has_trait = vampirism }
      is_ai = yes
    }
	}

  is_valid_showing_failures_only = {
    scope:recipient = {
      is_diplomatically_available = yes
    }
    custom_description = {
      text = recently_fed
      object = scope:actor
      NOT = {
        scope:actor = {
          has_character_modifier = blood_drink_well_fed
        }
      }
    }
  }  
    
	on_accept = {
    scope:actor = {
      add_stress = monumental_stress_gain
      stress_impact = {
        sadistic = medium_stress_impact_loss
      }
      add_character_modifier = {
        modifier = blood_kiss_aftermath_modifier
        days = 365
      }
      add_character_modifier = {
        modifier = blood_drink_well_fed
        days = 14
      }
    }
    scope:recipient = {
      add_character_flag = immune_to_disease
      add_trait = vampirism
      add_opinion = {
        modifier = sired_me_opinion
        target = scope:actor
      }
    }
  }
    
	ai_will_do = { base = 0 }
}
